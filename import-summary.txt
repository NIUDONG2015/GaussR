* libs\armeabi-v7a\libspeex.so => app\src\main\jniLibs\armeabi-v7a\libspeex.so
* libs\armeabi\libspeex.so => app\src\main\jniLibs\armeabi\libspeex.so
* res\ => app\src\main\res\
* src\ => app\src\main\java\
